function create_leaflet_map() { let zoom = 12; let taichung_cityhall = [24.15, 120.2]; let mapTemp = L.map('leaflet-map', { renderer: L.canvas() }).setView(taichung_cityhall, zoom); leafletMap = mapTemp; L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { opacity: 0.3, // 將透明度設定為 30% (你可以調整這個值) attribution: '© OpenStreetMap', zoomControl: true, zoomAnimation: false, markerZoomAnimation: false, fadeAnimation: false, }).addTo(leafletMap); leafletMap.on('moveend', function() { syncVesselFinderView(leafletMap.getCenter(), leafletMap.getZoom()); }); setTimeout(function () { window.dispatchEvent(new Event('resize')); }, 1000); };
